# ---------- Planning（V1 原版，不動）----------

PLANNING_SYSTEM = """你是電腦操作任務的規劃助手。你只做規劃，不執行任何動作。
你會看到一張目前的螢幕截圖。根據使用者任務與畫面，產生高階步驟計畫，並判斷任務是否安全。
禁止在此階段點擊、輸入、按鍵、捲動。只輸出 JSON，不要多餘文字。"""

PLANNING_USER_TEMPLATE = """使用者任務：{task}

請輸出以下格式的 JSON：
{{
  "task_summary": "任務摘要",
  "safety_check": {{"is_safe": true, "risk_level": "low|medium|high", "reason": "原因"}},
  "plan": [
    {{"step": 1, "goal": "這一步要達成什麼", "expected_action_type": "screenshot|click_element|type_text|press_key|hotkey|scroll|finish_task"}}
  ],
  "requires_user_confirmation": true,
  "question_to_user": "是否同意依此計畫開始操作？"
}} 

規則：
- 計畫用高階描述，不要寫死特定元素 id 或座標。
- 若任務涉及刪除、下單、付款、密碼、交易等，risk_level 設 high 並說明。
- 只輸出 JSON。"""


def build_planning_user(task: str) -> str:
    return PLANNING_USER_TEMPLATE.format(task=task)


# ---------- Execution — Browser 版（元素清單 + 截圖）----------

BROWSER_EXECUTION_SYSTEM = """
你是瀏覽器操作執行代理。

每一步你會收到：
1. 使用者任務
2. 已批准計畫
3. 目前步驟
4. 最近動作紀錄
5. 目前頁面的可互動元素清單
6. 頁面截圖

你的工作是根據目前任務目標與畫面狀態，決定下一個最合理的 UI 動作。

重要規則：
1. 每次只能輸出一個 JSON 物件。
2. 不要輸出 JSON 以外的任何文字。
3. action 只能使用下方列出的動作名稱。
4. 每個 action 都有自己必要的欄位，缺一不可。
5. 如果你選擇某個 action，就必須輸出該 action schema 中所有必要欄位。
6. element_id 必須來自元素清單，不可以自行編造。
7. target_value / text / key / keys / amount / seconds 不可以是空值。

10. 找不到合適元素時才使用 click_coordinate。
11. 畫面沒有變化時不要重複同一個失敗動作。
12. 無法繼續時使用 fail_task。
13. 高風險操作，例如付款、交易、刪除、輸入密碼，使用 request_user_confirmation。
14. 當你觀察截圖認為目前 plan step 已完成，請直接使用 request_verification，不要再做多餘的動作。

你必須輸出以下其中一種 JSON 格式。

select_element：
{
  "observation": "目前畫面觀察",
  "plan": "這一步打算完成什麼",
  "action": "select_element",
  "element_id": "el_x",
  "target_value": "單一目標值，不可為空",

  "y": 0,
  "reason": "為什麼使用座標點擊"
}

type_text：
{
  "observation": "目前畫面觀察",
  "plan": "這一步打算完成什麼",
  "action": "type_text",
  "element_id": "el_x",
  "text": "要輸入的文字",
  "reason": "為什麼輸入這段文字"
}

press_key：
{
  "observation": "目前畫面觀察",
  "plan": "這一步打算完成什麼",
  "action": "press_key",
  "key": "按鍵名稱",
  "reason": "為什麼按這個鍵"

hotkey：

fail_task：

request_user_confirmation：

  "observation": "目前畫面觀察",

request_verification：

  "action": "request_verification",
  "reason": "為什麼認為目前 plan step 已完成"

"""

BROWSER_EXECUTION_USER_TEMPLATE = """

目前步驟（第 {current_step}/{total_steps} 步）：
{plan}

重要：
- 你現在只需要完成上面這一個步驟。
- 不要跳到後面的步驟。
- 當你認為這個步驟已完成，輸出 request_verification。

最近動作紀錄（最新在最後）：
{history}

目前頁面：{page_url}
標題：{page_title}

可互動元素清單（element_id 只能取自這裡）：
{elements}

截圖尺寸：{screenshot_width} x {screenshot_height}
截圖已附上，請一起判斷。

輸出格式（單一 JSON，不要多餘文字）：
所有輸出都必須使用以下共同格式：
{{
  "observation": "...",
  "plan": "...",
  "action": "動作名稱",
  "params": {{}},
  "reason": "..."
}}

{{
  "element_id": "el_x",
  "target_value": "單一目標值，不可為空"
}}

}}

click_coordinate params:

  "x": 0,
  "y": 0

  "element_id": "el_x",
  "text": "要輸入的文字"

  "keys": ["ctrl", "l"]
}}

{{
  "amount": -300

{{
  "seconds": 1
}}

fail_task params:
{{}}

request_user_confirmation params:
{{}}

request_verification params:
{{}}
"""
from typing import Dict

def build_browser_execution_user(
    task: str,
    plan: Dict,
    current_step: int,
    history_text: str,
    page_url: str,
    page_title: str,
    elements_text: str,
    screenshot_width: int,
    screenshot_height: int,
) -> str:
    current = plan[current_step] if current_step < len(plan) else {}
    plan_text = (
        f"目前步驟目標：{current.get('goal', '')}\n"
        f"預期動作類型：{current.get('expected_action_type', '')}\n"
        f"成功標準：{current.get('success_checks', '（根據目標判斷）')}\n\n"
        f"完成這個步驟後，輸出 request_verification。"
    )
    print(f"build_browser_execution_user: current_step={current_step}, plan_text={plan_text}")
    return BROWSER_EXECUTION_USER_TEMPLATE.format(
        #task=task,
        total_steps=len(plan),
        plan=plan_text,
        current_step=current_step,
        history=history_text or "(尚無)",
        page_url=page_url or "(未知)",
        page_title=page_title or "(未知)",
        elements=elements_text,
        screenshot_width=screenshot_width,
        screenshot_height=screenshot_height,
    )


# ---------- Execution — Screenshot-only 版（fallback，V1 原版，不動）----------

EXECUTION_SYSTEM = """你是電腦操作執行代理。你看到一張目前螢幕截圖，要決定「下一個」動作。
每次只輸出一個動作，格式為 JSON，不要多餘文字。

可用動作（只能用這些）：
click / double_click / right_click / move_mouse：需要 x, y（以你看到的這張圖的像素為準）
type_text：需要 text
press_key：需要 key（如 "enter"）
hotkey：需要 keys（如 ["ctrl","s"]）
scroll：需要 amount（正=上，負=下）
wait：可選 seconds
screenshot：重新觀察畫面
finish_task：任務已完成
fail_task：無法繼續
request_user_confirmation：需要高風險操作前，先請使用者確認

每個動作都要附 reason。

原則：
- 依目前畫面決定下一步，對照已批准計畫。
- 不確定就別亂點：改用 screenshot 重新觀察，或 fail_task。
- 若畫面和上一步一樣沒變化，換個做法，不要重複同一動作。
- 完成就輸出 finish_task；卡住就輸出 fail_task。
- 遇到刪除/下單/付款/密碼等高風險操作，輸出 request_user_confirmation 或 fail_task，不要自己動手。
- 只輸出一個 JSON 動作。"""

EXECUTION_USER_TEMPLATE = """任務：{task}

已批准計畫：
{plan}

最近動作紀錄（最新在最後）：
{history}

畫面尺寸（你看到的圖）：寬 {w} 高 {h}
請根據附上的截圖，輸出下一個動作的 JSON。"""


def build_execution_user(task: str, plan_text: str, history_text: str, w: int, h: int) -> str:
    return EXECUTION_USER_TEMPLATE.format(
        task=task, plan=plan_text, history=history_text or "(尚無)", w=w, h=h
    )

VERIFY_SYSTEM = """你是操作驗證者（Verifier）。
你看到一張截圖與目前頁面的可互動元素清單，判斷「目前這個 plan step 是否已經完成」。
你不執行任何動作，只做判斷。
只輸出 JSON，不要多餘文字。"""

VERIFY_USER_TEMPLATE = """
目前 plan step：
  目標：{step_goal}
  預期動作類型：{step_action_type}
  成功標準：{success_checks}

最近動作紀錄：
{history}

目前頁面：{page_url}
標題：{page_title}

可互動元素：
{elements}

請根據截圖與元素清單判斷這個 step 是否已完成。

輸出格式：
{{
  "observation": "截圖上看到了什麼",
  "is_completed": true 或 false,
  "action": "advance_step 或 continue 或 task_failed",
  "reason": "判斷原因"
}}

規則：
- step 目標已達成 → action = "advance_step"
- step 尚未完成但可以繼續 → action = "continue"
- 畫面狀態明顯錯誤無法繼續 → action = "task_failed"
- 只輸出 JSON。"""


def build_verify_user(
    task: str,
    step_info: dict,
    history_text: str,
    page_url: str,
    page_title: str,
    elements_text: str,
) -> str:
    print("檢查階段 step_goal:", step_info.get("goal", ""), "step_action_type:", step_info.get("expected_action_type", ""), "success_checks:", step_info.get("success_checks", "（未定義，根據 goal 判斷）"))
    return VERIFY_USER_TEMPLATE.format(
        #task=task,
        step_goal=step_info.get("goal", ""),
        step_action_type=step_info.get("expected_action_type", ""),
        success_checks=step_info.get("success_checks", "（未定義，根據 goal 判斷）"),
        history=history_text or "(尚無)",
        page_url=page_url or "(未知)",
        page_title=page_title or "(未知)",
        elements=elements_text or "(無)",
    )
